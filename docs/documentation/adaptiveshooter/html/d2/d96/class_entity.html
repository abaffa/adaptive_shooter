unsigned int Entity::boundToScreen ( )

Bounds the entity to screen, canceling movement that would otherwise make the entity outside of screen boundaries

Returns:
unsigned int Directions where a collision has occurred. Check CollisionDirection for values. 
{
        unsigned int collision = CD_NOCOLLISION;
        CL_GraphicContext& gc = GameManager::getInstance()->getWindow()->get_gc();
        float top, bottom, left, right;

        top = _position.y;
        left = _position.x;
        right = left + _currentSprite->get_width();
        bottom = top + _currentSprite->get_height();

        if (bottom > gc.get_height())
        {
                _position.y = (float)(gc.get_height() - _currentSprite->get_height());
                collision |= CD_BOTTOM;
        } 
        
        if (right > gc.get_width())
        {
                _position.x = (float)(gc.get_width() - _currentSprite->get_width());
                collision |= CD_RIGHT;
        }
        
        if (top < 0.0f)
        {
                _position.y = 0.0f;
                collision |= CD_TOP;
        }
        
        if (left < 0.0f)
        {
                _position.x = 0.0f;
                collision |= CD_LEFT;
        }

        return collision;
}

bool Entity::checkWindowBoundary ( )

Checks if current sprite is still visible in screen boundaries

Returns:
bool true if inside graphics context window, false otherwise 
{
        CL_GraphicContext& gc = GameManager::getInstance()->getWindow()->get_gc();
        float top, bottom, left, right;

        top = _position.y;
        left = _position.x;
        right = left + _currentSprite->get_width();
        bottom = top + _currentSprite->get_height();

        if (bottom < 0.0f || right < 0.0f || top > gc.get_height() || left > gc.get_width())
        {
                return false;
        } 
        else
        {
                return true;
        }
}
