\end{DoxyReturn}

\begin{DoxyCode}
{
        std::vector<Shot*> shots;

        if (_canShoot)
        {
                TestScenePlayer *currentScene = dynamic_cast<TestScenePlayer*>( 
      GameManager::getInstance()->peekScene() );

                if (currentScene != NULL )
                {
                        Shot* newShot = new Shot(entity->getPosition().x + entity
      ->getCurrentSprite()->get_width()*0.5f,
                                entity->getPosition().y, _shotSpeed.x, _shotSpeed
      .y, _shotResource, _damage);
                        newShot->setPositionX( newShot->getPosition().x - newShot
      ->getCurrentSprite()->get_width()*0.5f );
                        shots.push_back( newShot );

                        setCanShoot( false );
                }
        }

        return shots;
}
